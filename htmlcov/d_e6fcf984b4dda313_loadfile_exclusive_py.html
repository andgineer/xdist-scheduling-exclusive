142 def check_schedule(self, node, duration=0): 

143 """Maybe schedule new items on the node 

144 

145 If there are any globally pending nodes left then this will 

146 check if the given node should be given any more tests. The 

147 ``duration`` of the last test is optionally used as a 

148 heuristic to influence how many tests the node is assigned. 

149 """ 

150 if node.shutting_down: 

151 return 

152 

153 if self.pending: 

154 # how many nodes do we have? 

155 num_nodes = len(self.node2pending) 

156 # if our node goes below a heuristic minimum, fill it out to 

157 # heuristic maximum 

158 items_per_node_min = max(2, len(self.pending) // num_nodes // 4) 

159 items_per_node_max = max(2, len(self.pending) // num_nodes // 2) 

160 node_pending = self.node2pending[node] 

161 if len(node_pending) < items_per_node_min: 

162 if duration >= 0.1 and len(node_pending) >= 2: 

163 # seems the node is doing long-running tests 

164 # and has enough items to continue 

165 # so let's rather wait with sending new items 

166 return 

167 num_send = items_per_node_max - len(node_pending) 

168 # keep at least 2 tests pending even if --maxschedchunk=1 

169 maxschedchunk = max(2 - len(node_pending), self.maxschedchunk) 

170 self._send_tests(node, min(num_send, maxschedchunk)) 

171 else: 

172 node.shutdown() 

173 

174 self.log("num items waiting for node:", len(self.pending))

176 def remove_node(self, node): 

177 """Remove a node from the scheduler 

178 

179 This should be called either when the node crashed or at 

180 shutdown time. In the former case any pending items assigned 

181 to the node will be re-scheduled. Called by the 

182 ``DSession.worker_workerfinished`` and 

183 ``DSession.worker_errordown`` hooks. 

184 

185 Return the item which was being executing while the node 

186 crashed or None if the node has no more pending items. 

187 

188 """ 

189 pending = self.node2pending.pop(node) 

190 if not pending: 

191 return 

192 

193 # The node crashed, reassing pending items 

194 crashitem = self.collection[pending.pop(0)] 

195 self.pending.extend(pending) 

196 for node in self.node2pending: 

197 self.check_schedule(node) 

198 return crashitem

200 def schedule(self): 

201 """Initiate distribution of the test collection 

202 

203 Initiate scheduling of the items across the nodes. If this 

204 gets called again later it behaves the same as calling 

205 ``.check_schedule()`` on all nodes so that newly added nodes 

206 will start to be used. 

207 

208 This is called by the ``DSession.worker_collectionfinish`` hook 

209 if ``.collection_is_completed`` is True. 

210 """ 

211 assert self.collection_is_completed 

212 

213 # Initial distribution already happened, reschedule on all nodes 

214 if self.collection is not None: 

215 for node in self.nodes: 

216 self.check_schedule(node) 

217 return 

218 

219 # XXX allow nodes to have different collections 

220 if not self._check_nodes_have_same_collection(): 

221 self.log("**Different tests collected, aborting run**") 

222 return 

223 

224 # Collections are identical, create the index of pending items. 

225 self.collection = list(self.node2collection.values())[0] 

226 self.pending[:] = range(len(self.collection)) 

227 if not self.collection: 

228 return 

229 

230 if self.maxschedchunk is None: 

231 self.maxschedchunk = len(self.collection) 

232 

233 # Custom scheduling for exclusive tests on dedicated nodes 

234 for node in self.dedicated_test_nodes: 

235 self._send_tests(node, 1) 

236 

237 # Adjust pending tests after exclusive tests have been scheduled 

238 if len(self.pending) < 2 * len(self.regular_nodes): 

239 nodes = cycle(self.regular_nodes) 

240 for i in range(len(self.pending)): 

241 self._send_tests(next(nodes), 1) 

242 else: 

243 # Send batches of consecutive tests. By default, pytest sorts tests 

244 # in order for optimal single-threaded execution, minimizing the 

245 # number of necessary fixture setup/teardown. Try to keep that 

246 # optimal order for every worker. 

247 

248 # how many items per node do we have about? 

249 items_per_node = len(self.collection) // len(self.node2pending) 

250 # take a fraction of tests for initial distribution 

251 node_chunksize = min(items_per_node // 4, self.maxschedchunk) 

252 node_chunksize = max(node_chunksize, 2) 

253 # and initialize each node with a chunk of tests 

254 for node in self.nodes: 

255 self._send_tests(node, node_chunksize) 

256 

257 if not self.pending: 

258 # initial distribution sent all tests, start node shutdown 

259 for node in self.nodes: 

260 node.shutdown()

262 def _send_tests(self, node, num): 

263 tests_for_node = [] 

264 

265 # First, check if the node is dedicated for exclusive tests 

266 if node in self.dedicated_test_nodes: 

267 # Attempt to find the first exclusive test for dedicated nodes 

268 for index, test in enumerate(self.pending): 

269 if self.collection[test] in self.exclusive_tests: 

270 # Schedule the found exclusive test and remove it from pending 

271 tests_for_node = [self.pending.pop(index)] 

272 break # Exit after scheduling the first found exclusive test 

273 # If no exclusive tests are pending, and only then, schedule a non-exclusive test 

274 if not tests_for_node: 

275 tests_for_node = self.pending[:num] 

276 del self.pending[:num] 

277 else: 

278 # For non-dedicated nodes, find up to num non-exclusive tests 

279 non_exclusive_tests = [test for test in self.pending if self.collection[test] not in self.exclusive_tests][:num] 

280 tests_for_node.extend(non_exclusive_tests) 

281 for test in non_exclusive_tests: 

282 self.pending.remove(test) # Remove the scheduled tests from pending 

283 

284 if tests_for_node: 

285 self.node2pending[node].extend(tests_for_node) 

286 node.send_runtest_some(tests_for_node)

288 def _check_nodes_have_same_collection(self): 

289 """Return True if all nodes have collected the same items. 

290 

291 If collections differ, this method returns False while logging 

292 the collection differences and posting collection errors to 

293 pytest_collectreport hook. 

294 """ 

295 node_collection_items = list(self.node2collection.items()) 

296 first_node, col = node_collection_items[0] 

297 same_collection = True 

298 for node, collection in node_collection_items[1:]: 

299 msg = report_collection_diff(col, collection, first_node.gateway.id, node.gateway.id) 

300 if msg: 

301 same_collection = False 

302 self.log(msg) 

303 if self.config is not None: 

304 rep = CollectReport(node.gateway.id, "failed", longrepr=msg, result=[]) 

305 self.config.hook.pytest_collectreport(report=rep) 

306 

307 return same_collection

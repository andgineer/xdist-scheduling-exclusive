14class ExclusiveLoadScopeScheduling(LoadScopeScheduling): # type: ignore # pylint: disable=abstract-method 

15 """Custom xdist scheduling. 

16 

17 Schedule tests from exclusive_tests.txt first and on dedicated nodes. 

18 Other tests are grouped as in `--dist loadfile`: tests from the same file run on the same node. 

19 """

22 self, 

23 config: Any,

25 exclusive_tests: Optional[List[str]] = None, 

26 dedicate_nodes: bool = False, 

27 ) -> None: 

28 """Load tests from exclusive_tests.txt. 

29 

30 If dedicate_nodes is True, exclusive tests exclusively occupy their nodes. 

31 """ 

32 super().__init__(config, log) 

33 self.exclusive_tests = exclusive_tests or load_exclusive_tests() 

34 self.dedicate_nodes = dedicate_nodes 

35 self.exclusive_tests_nodes: Set[str] = set() 

36 self.exclusive_tests_scheduled: Set[str] = set() 

37 

38 trace(f"LoadFileExclusiveScheduling have loaded {len(self.exclusive_tests)} exclusive tests.") 

39 self.dedicated_nodes_assigned = False

41 @property 

42 def collection_is_completed(self) -> bool: 

43 """Verify we have enough nodes for dedicated exclusive tests run.""" 

44 result = super().collection_is_completed 

45 if result: 

46 assert not self.dedicate_nodes or len(self.exclusive_tests_nodes) < self.numnodes, ( 

47 f"Not enough nodes ({self.numnodes}) to dedicate " 

48 f"to exclusive tests ({len(self.exclusive_tests_nodes)})" 

49 ) 

50 return result # type: ignore

52 def _assign_work_unit(self, node: Any) -> None: 

53 # First, attempt to assign exclusive tests if any are unscheduled 

54 exclusive_tests_to_schedule = set(self.exclusive_tests) - self.exclusive_tests_scheduled 

55 if exclusive_tests_to_schedule: 

56 for scope, work_unit in self.workqueue.items(): 

57 # Find if any test in the current scope is exclusive and unscheduled 

58 exclusive_test_in_scope = any(test in self.exclusive_tests for test in work_unit) 

59 if exclusive_test_in_scope: 

60 # Schedule the exclusive test 

61 self._schedule_exclusive_test(node, scope, work_unit) 

62 return # Exit after scheduling an exclusive test to ensure prioritization 

63 

64 if not self.dedicate_nodes or node.gateway.id not in self.exclusive_tests_nodes: 

65 super()._assign_work_unit(node)

67 def _schedule_exclusive_test(self, node: Any, scope: str, work_unit: Any) -> None: 

68 self.exclusive_tests_nodes.add(node.gateway.id) 

69 self.exclusive_tests_scheduled.update(work_unit.keys()) 

70 self.workqueue.pop(scope) 

71 self.assigned_work[node][scope] = work_unit 

72 self._send_work_to_node(node, work_unit) 

73 

74 def _send_work_to_node(self, node: Any, work_unit: Any) -> None: 

75 """Send work to the node. 

76 

77 The work_unit is a dictionary where the keys are test identifiers. 

78 This method converts those identifiers into the format expected by the node 

79 and then dispatches the work. 

80 """ 

81 test_identifiers = list(work_unit.keys()) 

82 

83 node_test_collection = self.registered_collections[node] 

84 test_indices = [ 

85 node_test_collection.index(test_id) 

86 for test_id in test_identifiers 

87 if test_id in node_test_collection 

88 ] 

89 

90 if test_indices: 

91 node.send_runtest_some(test_indices) 

92 trace(f"Sent {len(test_indices)} tests to {node} for execution.") 

93 else: 

94 trace(f"No matching tests found in node's collection for {node}.")

96 def _split_scope(self, nodeid: str) -> str: 

97 """Group tests by file, except for exclusive tests scheduled on dedicated nodes.""" 

98 if nodeid in self.exclusive_tests: 

99 # Treat each exclusive test as a unique scope 

100 return f"{EXCLUSIVE_TEST_SCOPE_PREFIX}::{nodeid}" 

101 return nodeid.split("::", 1)[0]
